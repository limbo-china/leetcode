/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
     ListNode *left, *right;
     left=head;
     if(!left||!left->next)
        return head;
     right=left->next;
     return head;   
    }
    
};
